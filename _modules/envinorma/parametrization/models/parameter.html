[docs]def dump_parameter_value(value: Any, type_: ParameterType) -> Any:
    if type_ == ParameterType.DATE:
        if isinstance(value, datetime):
            value = value.date()
        return str(value)
    if type_ == ParameterType.REGIME:
        return value.value
    return value




[docs]def load_parameter_value(json_value: Any, type_: ParameterType) -> Any:
    if type_ == ParameterType.DATE:
        if isinstance(json_value, int):
            date_ = datetime.fromtimestamp(json_value).date()
        else:
            date_ = date.fromisoformat(json_value)
        return date_
    if type_ == ParameterType.REGIME:
        return Regime(json_value)
    return json_value
